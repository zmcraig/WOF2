<!DOCTYPE html><html lang="en"><head><meta charset="UTF-8"><meta name="viewport" content="width=device-width, initial-scale=1.0"><title>Wheel of Fortune Simulator</title><style>    body { font-family: Arial, sans-serif; }    #puzzle, #category, #score { font-size: 24px; margin-bottom: 10px; }    #alphabet { margin-top: 20px; }    .letter { display: inline-block; width: 20px; padding: 5px; margin: 2px; background-color: #f0f0f0; text-align: center; }    #message { margin-top: 20px; color: red; font-size: 20px; }    input[type="text"], button { margin-top: 10px; }</style></head><body><h1>Wheel of Fortune Simulator</h1><div id="category"></div><div id="puzzle"></div><div id="score">Score: 0</div><input type="text" id="guess" placeholder="Enter a letter or solve the puzzle" /><button onclick="makeGuess()">Guess</button><div id="message"></div><div id="alphabet"></div><script>    const categories = ["Around the house", "Before & After", "Event", "Fun & Games", "Food & Drink",        "In the Kitchen", "Living Thing", "Occupation", "Person or People", "Phrase", "Place",         "Quotation", "Rhyme Time", "Same Letter", "Same Name", "Song Lyrics", "Thing", "Crossword",        "On the Map", "Toss Ups", "What Are You Doing?", "What Are You Wearing?"];    const phrases = {        "Teacher Appreciation Week": ["TEACHER APPRECIATION WEEK", "EDUCATIONAL JOURNEY", "CLASSROOM MANAGEMENT"],        "Back to School": ["SCHOOL SUPPLY SHOPPING", "MEETING NEW STUDENTS", "FIRST DAY OF SCHOOL"],        "Summer Vacation": ["SUMMER SCHOOL SESSION", "READING ASSIGNMENTS", "FIELD TRIPS PLANNING"]    };    let currentCategory = categories[Math.floor(Math.random() * categories.length)];    let currentPhrase = phrases[currentCategory][Math.floor(Math.random() * phrases[currentCategory].length)];    let displayedPhrase = currentPhrase.replace(/[^ ]/g, '_');    let score = 0;    let guessedLetters = [];    function updateDisplay() {        document.getElementById('category').textContent = 'Category: ' + currentCategory;        document.getElementById('puzzle').textContent = displayedPhrase.split('').join(' ');        document.getElementById('score').textContent = 'Score: ' + score;        updateAlphabet();    }    function updateAlphabet() {        const alphabet = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';        const alphabetDisplay = alphabet.split('').map(letter =>             `<span class="letter" style="${guessedLetters.includes(letter) ? 'background-color: #d3d3d3;' : ''}">${letter}</span>`        ).join('');        document.getElementById('alphabet').innerHTML = alphabetDisplay;    }    function makeGuess() {        const guess = document.getElementById('guess').value.toUpperCase();        document.getElementById('guess').value = ''; // Clear input after guess        if (guessedLetters.includes(guess) || guess.length === 0) {            document.getElementById('message').textContent = 'Already guessed or invalid input!';            return;        }        guessedLetters.push(guess);        if (guess.length == 1 && currentPhrase.includes(guess)) {            // Update the score and reveal letters            score += 100; // Score per correct letter guessed            displayedPhrase = displayedPhrase.split('').map((char, idx) => currentPhrase[idx] === guess ? guess : char).join('');            updateDisplay();        } else if (guess === currentPhrase) {            score += 1000; // Bonus for solving the puzzle            document.getElementById('message').textContent = 'Congratulations! You solved the puzzle!';            updateDisplay();        } else {            document.getElementById('message').textContent = 'Incorrect! Try again.';            score -= 100; // Penalty for wrong guess or solve attempt            updateDisplay();        }        checkVowels();    }    function checkVowels() {        const vowels = 'AEIOU';        let remainingVowels = vowels.split('').some(vowel => currentPhrase.includes(vowel) && !guessedLetters.includes(vowel));        if (!remainingVowels) {            document.getElementById('message').textContent += ' No more vowels left!';        }    }    updateDisplay();</script></body></html>
